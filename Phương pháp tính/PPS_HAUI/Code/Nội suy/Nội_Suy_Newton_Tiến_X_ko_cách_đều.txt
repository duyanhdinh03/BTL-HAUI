double process(int n, double x[], double y[], double X){
    double f[n + 1][n + 1];
    for(int i = 0;i<n;i++){
        f[i][0] = y[i];
    }
    for(int j = 1;j<n;j++){
        for(int i = 0;i<n-j;i++){
            f[i][j] = (f[i+1][j-1] - f[i][j-1]) / (x[i+j] - x[i]);
        }
    }
    double sum = 0;
    for(int i = 0;i<n;i++){
        double tmp = f[0][i];
        for(int j = 0;j<i;j++){
            tmp *= (X - x[j]);
        }
        sum += tmp;
    }
    return sum;
}
